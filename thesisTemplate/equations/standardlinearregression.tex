\begin{equation}
\label{eq:standardlinearregression}
\begin{align*}
y(\mathbf{x, w}) &= w_0 + w_1 x_1 + \ldots + w_n x_n \\
				 &= w_0 + \sum_{j = 1}^{M - 1}{w_j \mathbf{x_j}} \\
				 &= \sum_{j = 0}^{M - 1}{w_j \mathbf{x_j}} \\
				 &= \mathbf{w}^T \mathbf{x} .
\end{align*}
\end{equation}
